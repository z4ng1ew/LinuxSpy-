chmod +x main.sh


# Успешный случай
./main.sh Привет
# → Привет

# Пустой аргумент
./main.sh ""
# → error: заклинание не может быть пустым.

# Число
./main.sh 42
# → error: заклинание не должно быть числом.

# Нет аргумента
./main.sh
# → error: требуется ровно одно заклинание.

# Два аргумента
./main.sh раз два
# → error: требуется ровно одно заклинание.
