chmod +x whisper_validator.sh echo_spell.sh


# Успешный случай
./echo_spell.sh Привет
# → Привет

# Пустой аргумент
./echo_spell.sh ""
# → error: заклинание не может быть пустым.

# Число
./echo_spell.sh 42
# → error: заклинание не должно быть числом.

# Нет аргумента
./echo_spell.sh
# → error: требуется ровно одно заклинание.

# Два аргумента
./echo_spell.sh раз два
# → error: требуется ровно одно заклинание.
